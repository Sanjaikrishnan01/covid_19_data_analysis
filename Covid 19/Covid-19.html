import matplotlib.pyplot as plt
import seaborn as sns

plt.figure(figsize=(12, 8))
sns.heatmap(df.corr(numeric_only=True), annot=True, cmap='coolwarm', fmt=".2f")
plt.title('Correlation Matrix of Numerical COVID-19 Data')
plt.show()

regional_summary = df.groupby('WHO Region')[['Confirmed', 'Deaths', 'Recovered']].sum().reset_index()

plt.figure(figsize=(15, 7))
bar_width = 0.25
r1 = range(len(regional_summary['WHO Region']))
r2 = [x + bar_width for x in r1]
r3 = [x + bar_width for x in r2]

plt.bar(r1, regional_summary['Confirmed'], color='skyblue', width=bar_width, edgecolor='grey', label='Confirmed')
plt.bar(r2, regional_summary['Deaths'], color='salmon', width=bar_width, edgecolor='grey', label='Deaths')
plt.bar(r3, regional_summary['Recovered'], color='lightgreen', width=bar_width, edgecolor='grey', label='Recovered')

plt.xlabel('WHO Region', fontweight='bold')
plt.ylabel('Total Cases', fontweight='bold')
plt.title('Total Confirmed, Deaths, and Recovered Cases by WHO Region')
plt.xticks([r + bar_width for r in range(len(regional_summary['WHO Region']))], regional_summary['WHO Region'], rotation=45, ha='right')
plt.legend()
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import numpy as np

metrics_to_plot = ['Confirmed', 'Deaths', 'Recovered', 'Active']
x = np.arange(len(selected_countries_df['Country/Region']))
bar_width = 0.18

plt.figure(figsize=(18, 8))

for i, metric in enumerate(metrics_to_plot):
    plt.bar(x + i * bar_width, selected_countries_df[metric], bar_width, label=metric)

plt.ylabel('Number of Cases')
plt.title('Comparison of Key COVID-19 Metrics Across Selected Countries')
plt.xticks(x + bar_width * (len(metrics_to_plot) - 1) / 2, selected_countries_df['Country/Region'], rotation=45, ha='right')
plt.legend()
plt.tight_layout()
plt.show()
